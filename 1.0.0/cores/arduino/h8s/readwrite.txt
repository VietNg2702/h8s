		PutChar(*buffer++);
}

int _read( int a, char *buffer, int iSize){
	char *buff;
	int bytes = 0;
	buff = buffer;

	do{
		*buff = GetChar();
		PutChar(*buff);
		if(*buff != 13) buff++;
	}while(*buff != 13);
	*(buff) = '\0';
	return 0;
}
